library(dplyr)
library(magrittr)
data("PakPC2017Tehsil")

PakPC2017Tehsil %>%
  group_by(Province, Division, District) %>%
  summarise(
    Pop1998 = sum(Pop1998, na.rm = TRUE)
  , Pop2017 = sum(Pop2017, na.rm = TRUE))

 PakPC2017Tehsil %>%
  filter(District == "Faisalabad")
