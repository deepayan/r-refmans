Blackjack R6 Class

Description

Blackjack R6 Class

Usage

Blackjack

Format

An object of class R6ClassGenerator of length 24.

Examples

set.seed(101315)
setup()

# sit at the blackjack table
x <- Blackjack$new(who = "Player 1", bet = 10)

# play a hand
x$play()

x$hit()

x$stand()

# play a hand blind w/out drawing
x$play()$stand()

# clean-up
delete()

# reset the deck
x$shuffle()
x

# create a deck composed of 5 decks
x <- Deck$new(decks = 5)
x

Poker R6 Class

Description

Poker R6 Class

Usage

Poker

Format

An object of class R6ClassGenerator of length 24.

Examples

set.seed(101315)
setup()

# draw poker
x <- Poker$new(who = "Player 1", type = "draw", bet = 10)
x$play()
x$hold(1, 2, 5)
x$draw()

# stud poker (bet 20)
x <- Poker$new(who = "Player 1", type = "stud", bet = 20)
x$play()

# clean-up
delete()
